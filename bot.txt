# -------- HELPERS --------
def calculate_duration(start, end):
    fmt = "%I:%M%p"
    t1 = datetime.strptime(start.upper(), fmt)
    t2 = datetime.strptime(end.upper(), fmt)
    if t2 <= t1:
        t2 += timedelta(days=1)
    return int((t2 - t1).total_seconds())

def get_target_size(minutes):
    if minutes <= 46: return 300
    if minutes <= 50: return 400
    if minutes <= 60: return 600
    if minutes <= 120: return 1200
    return None

def calculate_bitrate(size_mb, duration_sec):
    total_bitrate = (size_mb * 8192) / duration_sec
    audio_bitrate = 128
    return int(total_bitrate - audio_bitrate)

def verify_link(link: str):
    if "gplinks.com" in link:
        api_key = API_KEYS["gplinks.com"]
        resp = requests.get(f"https://gplinks.com/api?api={api_key}&url={link}")
        if resp.status_code == 200 and resp.json().get("success"):
            return "Verified ✅ (GPLINK)"
        else:
            return "Invalid ❌ (GPLINK)"
    elif "shortxlinks.in" in link:
        api_key = API_KEYS["shortxlinks.in"]
        resp = requests.get(f"https://shortxlinks.in/api?api={api_key}&url={link}")
        if resp.status_code == 200 and resp.json().get("success"):
            return "Verified ✅ (SHORTXLINK)"
        else:
            return "Invalid ❌ (SHORTXLINK)"
    else:
        return "Unknown Provider ❌"

# -------- COMMANDS --------

async def verify(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = update.effective_user.id
    username = update.effective_user.username or update.effective_user.first_name
    expiry = datetime.now() + timedelta(hours=VERIFY_DURATION_HOURS)
    users[user_id] = {"username": username, "verified_until": expiry}
    await update.message.reply_text(f"✅ {username} verified for {VERIFY_DURATION_HOURS} hours!")

async def statusme(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = update.effective_user.id
    info = users.get(user_id)
    if info:
        now = datetime.now()
        if now < info["verified_until"]:
            remaining = info["verified_until"] - now
            hours = remaining.seconds // 3600
            minutes = (remaining.seconds % 3600) // 60
            await update.message.reply_text(f"✅ Verified! Time left: {hours}h {minutes}m")
        else:
            users.pop(user_id)
            await update.message.reply_text("❌ Verification expired!")
    else:
        await update.message.reply_text("❌ You are not verified!")

async def add(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if len(context.args) < 2:
        await update.message.reply_text("Usage: /add <ID> <username>")
        return
    uid = int(context.args[0])
    uname = context.args[1]
    users[uid] = {"username": uname, "verified_until": datetime.now() + timedelta(hours=VERIFY_DURATION_HOURS)}
    await update.message.reply_text(f"✅ Added user {uname} with ID {uid}")

async def remove(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if len(context.args) < 1:
        await update.message.reply_text("Usage: /remove <ID or username>")
        return
    target = context.args[0]
    found = False
    try:
        uid = int(target)
        if uid in users:
            uname = users.pop(uid)["username"]
            await update.message.reply_text(f"❌ Removed user {uname} with ID {uid}")
            found = True
    except:
        pass
    if not found:
        for uid, info in list(users.items()):
            if info["username"].lower() == target.lower():
                users.pop(uid)
                await update.message.reply_text(f"❌ Removed user {info['username']} with ID {uid}")
                found = True
                break
    if not found:
        await update.message.reply_text(f"⚠ User '{target}' not found!")

async def checklink(update: Update, context: ContextTypes.DEFAULT_TYPE):

        await update.message.reply_text("Usage: /checklink <url>")
        return
    link = context.args[0]
    result = verify_link(link)
    await update.message.reply_text(result)

# -------- VIDEO QUEUE PROCESSING --------
async def process_queue(app):
    global active_recordings
    while True:
        if recording_queue and active_recordings < MAX_ACTIVE:
            task = recording_queue.pop(0)
            user_id, username, link, start_time, end_time, options = task
            active_recordings += 1

            duration = calculate_duration(start_time, end_time)
            minutes = duration // 60
            target_size = get_target_size(minutes)
            video_bitrate = calculate_bitrate(target_size, duration)
            output = f"{username}_{len(completed_recordings)+1}.mp4"

            vf_filters = ["scale=640:480"]
            if options.get("blackbars"): vf_filters.append("pad=854:480:(854-640)/2:0:black")
            if options.get("watermark"):
                vf_filters.append("drawtext=text='Anime-cartoon.kesug.com':fontcolor=white:fontsize=22:box=1:boxcolor=black@1:x=w-tw-40:y=35")
            if options.get("title"):
                vf_filters.append(f"drawtext=text='{options['title']}':fontcolor=yellow:fontsize=30:x=(w-text_w)/2:y=h-60:enable='between(t,{duration-5},{duration})'")
            vf_string = ",".join(vf_filters)

            cmd = [
                "ffmpeg",
                "-i", link,
                "-map", "0",
                "-t", str(duration),
                "-vf", vf_string,
                "-c:v", "libx264",
                "-preset", "fast",
                "-b:v", f"{video_bitrate}k",
                "-c:a", "aac",
                "-b:a", "128k"
            ]
            if options.get("audiorename"):
                cmd += ["-metadata:s:a:0", "title=[Anime-Cartoon.kesug.com] Hindi"]
            cmd += ["-movflags", "+faststart", output]

            subprocess.run(cmd)
            completed_recordings.append((user_id, username, output))
            await app.bot.send_message(chat_id=user_id, text=f"✅ Recording {output} completed")
            active_recordings -= 1
        await asyncio.sleep(1)

# -------- MESSAGE HANDLER --------
async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = update.effective_user.id
    if user_id not in users:
        await update.message.reply_text("❌ You must /verify first!")
        return

    text = update.message.text
    lower = text.lower()
    times = re.findall(r"\b\d{1,2}:\d{2}\s?(?:am|pm)\b", lower)
    link_match = re.search(r"https?://\S+\.m3u8", text)

    if len(times) < 2 or not link_match:
        await update.message.reply_text("Wrong format. Example:\n10:30PM 10:50PM http://example.m3u8")
        return

    start_time = times[0].replace(" ", "")
    end_time = times[1].replace(" ", "")
    link = link_match.group()

    # Options parsing
    options = {
        "watermark": "watermark" in lower,
        "blackbars": "black bars" in lower,
        "audiorename": "audio rename" in lower
    }
    title_match = re.search(r"title (.+)", text, re.IGNORECASE)
    if title_match: options["title"] = title_match.group(1)

    # Add to queue
    recording_queue.append((user_id, users[user_id]["username"], link, start_time, end_time, options))
    await update.message.reply_text(f"✅ Added to recording queue. Position: {len(recording_queue)}")
